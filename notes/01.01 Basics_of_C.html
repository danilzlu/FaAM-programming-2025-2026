2. Printf Examples

#include <stdio.h>

int main() {
    // Базовый printf
    printf("This is a simple message\n");
    
    // Вывод переменных
    int number = 42;
    printf("The answer is %d\n", number);
    
    // Несколько переменных
    float pi = 3.14159;
    printf("Integer: %d, Float: %.2f\n", number, pi);
    
    // Параметры форматирования
    printf("Right-aligned: %10d\n", number);  // Width of 10
    printf("Left-aligned: %-10d\n", number);  // Выравнивание влево
    printf("Zero-padded: %05d\n", number);    // Дополнение нулями
    
    return 0;
}

3. Variables (Initialization)

    int a = 10;
    short int b = 20;
    long int c = 1000000L;
    unsigned int d = 50;
    
    // Типы с плавающей точкой

    printf("Unsigned int d = %u\n", d);
    printf("Float f = %f\n", f);
    printf("Double g = %.10f\n", g);
    printf("Character ch = %c\n", ch);
    
    return 0;
}

4. Assignment (Присваивание)

#include <stdio.h>

int main() {
    int a;
    float b;
    
    // Простое присваивание
    a = 10;
    b = 3.14;
    
    printf("a = %d, b = %.2f\n", a, b);
    
    // Множественное присваивание
    int x, y, z;
    x = y = z = 5;
    printf("x = %d, y = %d, z = %d\n", x, y, z);
    
    // Составное присваивание
    a += 5;  // Equivalent to a = a + 5
    b *= 2;  // Equivalent to b = b * 2
    printf("After compound assignment: a = %d, b = %.2f\n", a, b);
    
    return 0;
}

5. Arithmetic Operations

#include <stdio.h>

int main() {
    int a = 10, b = 3;
    float x = 10.0, y = 3.0;
    
    // Целочисленная арифметика
    printf("Integer operations:\n");
    printf("a + b = %d\n", a + b);  // Сложение
    printf("a - b = %d\n", a - b);  // Вычитание
    printf("a * b = %d\n", a * b);  // Умножение
    printf("a / b = %d\n", a / b);  // Деление (целочисленное)
    printf("a %% b = %d\n", a % b); // Остаток от деления
    
    // Арифметика с плавающей точкой
    printf("\nFloat operations:\n");
    printf("x + y = %.2f\n", x + y);
    printf("x - y = %.2f\n", x - y);
    printf("x * y = %.2f\n", x * y);
    printf("x / y = %.2f\n", x / y);  // Деление даёт дробный результат
    
    // Инкремент и декремент
    int c = 5;
    printf("\nOriginal c = %d\n", c);
    printf("c++ = %d\n", c++);  // Постинкремент: сначала использовать, потом увеличить
    printf("Now c = %d\n", c);
    printf("++c = %d\n", ++c);  // Преинкремент: сначала увеличить, потом использовать
    printf("Now c = %d\n", c);
    
    return 0;
}

6. Scanf (User Input)

#include <stdio.h>

int main() {
    int age;
    float height;
    char initial;
    
    // Ввод целого числа
    printf("Enter your age: ");
    scanf("%d", &age);
    
    // Ввод числа с плавающей точкой
    printf("Enter your height in meters: ");
    scanf("%f", &height);
    
    // Ввод символа
    printf("Enter your first initial: ");
    scanf(" %c", &initial);  // Обратите внимание на пробел перед %c, чтобы убрать пробелы
    
    // Вывод введённых данных
    printf("\nYour age is %d years\n", age);
    printf("Your height is %.2f meters\n", height);
    printf("Your initial is %c\n", initial);
    
    // Ввод нескольких значений
    int num1, num2;
    printf("\nEnter two integers separated by space: ");
    scanf("%d %d", &num1, &num2);
    printf("You entered: %d and %d\n", num1, num2);
    
    return 0;
}

7. If Statement

#include <stdio.h>

int main() {
    int number;
    
    printf("Enter a number: ");
    scanf("%d", &number);
    
    // Простой оператор if
    if (number > 0) {
        printf("The number is positive.\n");
    }
    
    // Другой пример
    if (number % 2 == 0) {
        printf("The number is even.\n");
    }
    
    // Логические операторы в условиях
    if (number > 0 && number < 10) {
        printf("The number is between 0 and 10.\n");
    }
    
    if (number < 0 || number > 100) {
        printf("The number is either negative or greater than 100.\n");
    }
    
    return 0;
}

8. If...Else Statement

#include <stdio.h>

int main() {
    int number;
    
    printf("Enter a number: ");
    scanf("%d", &number);
    
    // Оператор if-else
    if (number > 0) {
        printf("The number is positive.\n");
    } else {
        printf("The number is zero or negative.\n");
    }
    
    // Оператор if-else if-else
    if (number > 0) {
        printf("The number is positive.\n");
    } else if (number < 0) {
        printf("The number is negative.\n");
    } else {
        printf("The number is zero.\n");
    }
    
    // Вложенный if-else
    if (number != 0) {
        if (number % 2 == 0) {
            printf("The number is even and not zero.\n");
        } else {
            printf("The number is odd.\n");
        }
    } else {
        printf("The number is zero.\n");
    }
    
    // Тернарный оператор (условное выражение)
    printf("The number is %s.\n", (number % 2 == 0) ? "even" : "odd");
    
    return 0;
}

9. Arrays

#include <stdio.h>

int main() {
    // Объявление и инициализация
    int numbers[5] = {10, 20, 30, 40, 50};
    
    // Доступ к элементам массива
    printf("First element: %d\n", numbers[0]);
    printf("Third element: %d\n", numbers[2]);
    
    // Изменение элементов массива
    numbers[1] = 25;
    printf("Modified second element: %d\n", numbers[1]);
    
    // Цикл по массиву
    printf("\nAll array elements:\n");
    for (int i = 0; i < 5; i++) {
        printf("numbers[%d] = %d\n", i, numbers[i]);
    }
    
    // Сумма элементов массива
    int sum = 0;
    for (int i = 0; i < 5; i++) {
        sum += numbers[i];
    }
    printf("\nSum of all elements: %d\n", sum);
    
    // Двумерный массив (матрица)
    int matrix[3][3] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };
    
    printf("\n2D array (matrix):\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            printf("%d ", matrix[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}

10. While Loop

#include <stdio.h>

int main() {
    // Простой цикл while
    int count = 1;
    printf("Counting with while loop:\n");
    while (count <= 5) {
        printf("%d ", count);
        count++;
    }
    printf("\n");
    
    // Сумма чисел
    int sum = 0;
    int i = 1;
    while (i <= 10) {
        sum += i;
        i++;
    }
    printf("Sum of numbers 1 to 10: %d\n", sum);
    
    // Ввод до выполнения условия
    int number;
    printf("\nEnter numbers (enter 0 to stop):\n");
    scanf("%d", &number);
    while (number != 0) {
        printf("You entered: %d\n", number);
        printf("Enter another number (0 to stop): ");
        scanf("%d", &number);
    }
    
    // Цикл do-while (выполняется хотя бы один раз)
    int j = 1;
    printf("\nCounting with do-while loop:\n");
    do {
        printf("%d ", j);
        j++;
    } while (j <= 5);
    printf("\n");
    
    return 0;
}

11. For Loop

#include <stdio.h>

int main() {
    // Простой цикл for
    printf("Counting with for loop:\n");
    for (int i = 1; i <= 5; i++) {
        printf("%d ", i);
    }
    printf("\n");
    
    // Цикл с другим шагом
    printf("\nCounting by twos:\n");
    for (int i = 0; i <= 10; i += 2) {
        printf("%d ", i);
    }
    printf("\n");
    
    // Вложенные циклы for (таблица умножения)
    printf("\nMultiplication table (5x5):\n");
    for (int i = 1; i <= 5; i++) {
        for (int j = 1; j <= 5; j++) {
            printf("%3d ", i * j);  // %3d обеспечивает выравнивание
        }
        printf("\n");
    }
    
    // Обратный счёт
    printf("\nCounting backward:\n");
    for (int i = 10; i > 0; i--) {
        printf("%d ", i);
    }
    printf("\n");
    
    // Цикл for с несколькими переменными
    printf("\nMultiple counter variables:\n");
    for (int i = 0, j = 10; i <= 10 && j >= 0; i++, j--) {
        printf("i=%d, j=%d\n", i, j);
    }
    
    return 0;
}

12. Functions

#include <stdio.h>

// Function declaration
void greet();
int add(int a, int b);
int sumEvenNumbers(int limit);

int main() {
    // Простой вызов функции
    greet();
    
    // Функция с возвращаемым значением
    int sum = add(5, 3);
    printf("Sum: %d\n", sum);
    
    // Функция с if и циклом while
    int evenSum = sumEvenNumbers(10);
    printf("Sum of even numbers up to 10: %d\n", evenSum);
    
    return 0;
}

// Функция без параметров и возвращаемого значения
void greet() {
    printf("Hello, welcome to C functions!\n");
}

// Функция с параметрами и возвращаемым значением
int add(int a, int b) {
    return a + b;
}

// Функция с использованием if и while
int sumEvenNumbers(int limit) {
    int sum = 0;
    int i = 1;
    
    while (i <= limit) {
        if (i % 2 == 0) {
            sum += i;
        }
        i++;
    }
    
    return sum;
}

13. Structures

#include <stdio.h>
#include <string.h>

// Простое определение структуры
struct Student {
    int id;
    char name[50];
    float gpa;
};

// Функция, которая создаёт и возвращает структуру Student
struct Student createStudent(int id, const char *name, float gpa) {
    struct Student newStudent;
    newStudent.id = id;
    strcpy(newStudent.name, name);
    newStudent.gpa = gpa;
    return newStudent;
}

int main() {
    // Создание структуры с помощью функции
    struct Student student1 = createStudent(101, "John Doe", 3.75);
    
    // Создание другой структуры напрямую
    struct Student student2;
    student2.id = 102;
    strcpy(student2.name, "Jane Smith");
    student2.gpa = 3.90;
    
    // Доступ к элементам структуры
    printf("Student 1 ID: %d\n", student1.id);
    printf("Student 1 Name: %s\n", student1.name);
    printf("Student 1 GPA: %.2f\n\n", student1.gpa);
    
    printf("Student 2 ID: %d\n", student2.id);
    printf("Student 2 Name: %s\n", student2.name);
    printf("Student 2 GPA: %.2f\n", student2.gpa);
    
    return 0;
}
